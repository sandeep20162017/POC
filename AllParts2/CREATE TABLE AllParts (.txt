CREATE TABLE BCES.AllParts (
    PartID INT IDENTITY(1,1) PRIMARY KEY,
    MMSStockCode VARCHAR(30),
    CorePartNumber VARCHAR(30),
    PartType VARCHAR(50),
    JobNumber VARCHAR(30),
    DetailedDesc VARCHAR(500),
    DateEntered DATETIME,
    ItemRefNumber VARCHAR(16),
    OverheadType NUMERIC(9, 2),
    CoreCharge NUMERIC(9, 2),
    PartCost NUMERIC(9, 2),
    AddedBy VARCHAR(30),
    MMSNewCost NUMERIC(15, 2),
    MSSyncDate DATETIME,
    OrigSupplierNum VARCHAR(25),
    OrigSupplierName VARCHAR(40),
    LastModifiedBy VARCHAR(30),
    LastModifiedDate DATETIME
);
